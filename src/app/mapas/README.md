   ng g c mapas/components/miniMapa --skip-tests -is
   ng g c mapas/pages/fullScreen --skip-tests -is 
   ng g c mapas/pages/marcadores --skip-tests -is
   ng g c mapas/pages/zoomRange --skip-tests -is
